void DrawLineHorizon(CDC *pDC)
{
	double alt,az;
	double x,y,z,ox,oy;
	double ad;
	int fail;

	UsePen(pDC,DKGRAY);

	ad=HALFWIDTH/zoom;

	ox=-9999;oy=-1;
	for (int i=0;i<74;i++)
	{
		fail=0;

		alt=0;
		az=horizon[i].az;

		//alt-=window_alt;
		alt-=D2R(win_alt.get_all());

		y=sin(alt)*cos(az)*ad;
		x=sin(az)*ad;

		z=R2D(az);

		if ((z<90) || (z>270))
		{

			y=-y;

			if (ox!=-9999)		
				ClipAndDrawLine(pDC,x,y,ox,oy);
			ox=x;oy=y;
		}
		else
			ox=-9999;
	}

}

void DrawSolidHorizon(CDC *pDC)
{
	double alt,az;
	double x,y,z,ox,oy;
	double ad;
	int fail;

	POINT pt[4];

	UsePen(pDC,LTGRAY);
	UseBrush(pDC,LTGRAY);


	ad=HALFWIDTH/zoom;

	ox=-1;oy=-1;
	for (int i=0;i<74;i++)
	{
		fail=0;

		alt=0;
		az=horizon[i].az;

		alt+=D2R(win_alt.get_all());

		y=sin(alt)*cos(az)*ad;
		x=sin(az)*ad;

		z=R2D(az);

		if ((z<90) || (z>270))
		{

			if (ox!=-1)
			{
				FindClipLine(x,y,x,HALFHEIGHT,&pt[0].x,&pt[0].y,&pt[1].x,&pt[1].y);
				FindClipLine(ox,HALFHEIGHT,ox,oy,&pt[2].x,&pt[2].y,&pt[3].x,&pt[3].y);

				if ((pt[0].x!=-9999) && (pt[2].x!=-9999))
					{
						pt[0].x+=HALFWIDTH;pt[0].y+=HALFHEIGHT;
						pt[1].x+=HALFWIDTH;pt[1].y+=HALFHEIGHT;
						pt[2].x+=HALFWIDTH;pt[2].y+=HALFHEIGHT;
						pt[3].x+=HALFWIDTH;pt[3].y+=HALFHEIGHT;
						pDC->Polygon(pt,4);
					}
				else
				{
					FindClipLine(x,y,ox,oy,&pt[0].x,&pt[0].y,&pt[1].x,&pt[1].y);
					FindClipLine(ox,HALFHEIGHT,x,HALFHEIGHT,&pt[2].x,&pt[2].y,&pt[3].x,&pt[3].y);
					if ((pt[0].x!=-9999) && (pt[2].x!=-9999))
					{
						pt[0].x+=HALFWIDTH;pt[0].y+=HALFHEIGHT;
						pt[1].x+=HALFWIDTH;pt[1].y+=HALFHEIGHT;
						pt[2].x+=HALFWIDTH;pt[2].y+=HALFHEIGHT;
						pt[3].x+=HALFWIDTH;pt[3].y+=HALFHEIGHT;
						pDC->Polygon(pt,4);
					}
				}
			}
			ox=x;oy=y;
		}
		else
			ox=-1;
	}
}
